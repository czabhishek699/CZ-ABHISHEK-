Question 1: Write a C program to read name, roll and marks in 3 subjects and display average marks of subjects using structure.

//CZ ABHISHEK 2024
#include<stdio.h>
struct student{
    char name[20];
    float eng,nep,com,avg;
};
    int main(){
        struct student s1;
        printf("Enter Name: ");
        scanf("%s", s1.name);
        printf("Enter English Marks: ");
        scanf("%f", &s1.eng);
        printf("Enter Nepali Marks: ");
        scanf("%f", &s1.nep);
        printf("Enter Computer Marks: ");
        scanf("%f", &s1.com);
        s1.avg=(s1.eng+s1.nep+s1.com)/3;
        printf("Average marks of 3 subjects: %0.2f", s1.avg);
    return 0;
    }

Question 2: Write a C program that uses structure to represent details of five books (title, author, publisher, and price) and print them out.

//CZ ABHISHEK 2024
#include<stdio.h>
struct book {
    char title[20];
    char author[20];
    char publisher[20];
    int price;
};
int main() {
    struct book b[5];
    for(int a = 0; a < 5; a++) {  
        printf("For Book %d:\n", a+1);
        printf("Title of Book: ");
        scanf("%s", b[a].title);
        printf("Author of Book: ");
        scanf("%s", b[a].author);
        printf("Publisher of Book: ");
        scanf("%s", b[a].publisher);
        printf("Price of Book: ");
        scanf("%d", &b[a].price);
    }
  for(int a = 0; a < 5; a++) {
        printf("\nDetails of Book %d:\n", a+1);
        printf("Title of Book: %s\n", b[a].title);
        printf("Author of Book: %s\n", b[a].author);
        printf("Publisher of Book: %s\n", b[a].publisher);
        printf("Price of Book: %d\n", b[a].price);
    }
   return 0;
}

Question 3: If you were a developer in Global Tech Inc. Company and were given the task to develop a program using C structures to ask for the information of 12 students with roll number, name, and marks scored in 3 subjects. Also, display them in proper format along with the calculation of total and percentage.

//CZ ABHISHEK 2024
#include<stdio.h>
struct student {
    int roll;
    char name[30];
    float eng, nep, com, marks, percent;
};
int main() {
    struct student s[12];
    int i;
    for(i = 0; i < 12; i++) {
        printf("For student %d:\n", i+1);
        printf("Enter name of student: ");
        scanf("%s", s[i].name);
        printf("Enter Roll No.: ");
        scanf("%d", &s[i].roll);
        printf("Enter Marks of English, Nepali, Computer: ");
        scanf("%f %f %f", &s[i].eng, &s[i].nep, &s[i].com);
       s[i].marks = s[i].eng + s[i].nep + s[i].com;
        s[i].percent = (s[i].marks / 300.0) * 100;
    }
    for(i = 0; i < 2; i++) {
        printf("\nDetails of student %d:\n", i+1);
        printf("Name: %s\n", s[i].name);
        printf("Roll No.: %d\n", s[i].roll);
        printf("Total Marks: %.2f\n", s[i].marks);
        printf("Percentage: %.2f%%\n", s[i].percent);
    }
   return 0;
}
